import pandas as pd
import plotly.express as px
from ipywidgets import interact, Dropdown
from textblob import TextBlob

# 1. Load filtered data
df = pd.read_csv("fintech_reviews_filtered.csv", parse_dates=["review_datetime_utc"], dayfirst=True)

# 2. Recompute columns
df["sentiment"] = df["review_text"].fillna("").apply(lambda t: TextBlob(t).sentiment.polarity)
flags = {
    "kyc": r"\b(kyc|verify|verification|validate|validation|id)\b",
    "failure": r"\b(fail|failed|error|timeout|disconnect|declin)\b",
    "feature": r"\b(split|bill|invest|save(?:\s*goal)?)\b",
    "support": r"\b(chat|support|help|agent|ticket)\b",
    "churn": r"\b(uninstall|switch|delete|cancel)\b"
}
for name, patt in flags.items():
    df[f"{name}_flag"] = df["review_text"].str.contains(patt, case=False, na=False)

# 3. Aggregate metrics (using the actual column names)
metrics = (
    df
    .groupby("application_id")
    .agg({
        "review_rating":  "mean",
        "sentiment":      "mean",
        "kyc_flag":       "mean",
        "failure_flag":   "mean",   # <— use the real name here
        "feature_flag":   "mean",
        "support_flag":   "mean",
        "churn_flag":     "mean"
    })
    .reset_index()
    .rename(columns={
        "application_id": "App",
        "review_rating":  "Avg Rating",
        "sentiment":      "Avg Sentiment",
        "kyc_flag":       "% KYC Mentions",
        "failure_flag":   "% Failure Mentions",   # <— match this too
        "feature_flag":   "% Feature Mentions",
        "support_flag":   "% Support Mentions",
        "churn_flag":     "% Churn Intent"
    })
)

# Map app IDs to friendly names
app_names = {
    "com.globe.gcash.android": "GCash",
    "com.paymaya": "Maya",
    "ph.seabank.seabank": "SeaBank"
}
metrics["App"] = metrics["App"].map(app_names)


# Show the full metrics table
from IPython.display import display
display(metrics)

# Interactive bar chart
from ipywidgets import interact, Dropdown
import plotly.express as px

@interact(
    metric=Dropdown(

    )
)
def show_chart(metric):
    fig = px.bar(
        metrics,
        x="App",
        y=metric,
        title=f"{metric} by App",
        labels={"App": "App", metric: metric}
    )
    fig.show()

import plotly.graph_objects as go

apps = metrics["App"].tolist()
metrics_list = [c for c in metrics.columns if c != "App"]

fig = go.Figure()

# Add one bar trace per metric, only first visible
for i, metric in enumerate(metrics_list):
    fig.add_trace(go.Bar(
        x=apps,
        y=metrics[metric],
        name=metric,   # Legend for metric
        visible=(i == 0),
        marker_line_width=1
    ))

# Dropdown for metric selection
buttons = []
for i, metric in enumerate(metrics_list):
    visibility = [False] * len(metrics_list)
    visibility[i] = True
    buttons.append(dict(
        label=metric,
        method="update",
        args=[
            {"visible": visibility},
            {"title": {"text": f"{metric} by App", "x": 0.5},  # Center title
             "yaxis": {"title": metric},
             "showlegend": True
            }
        ]
    ))

fig.update_layout(
    updatemenus=[dict(
        type="dropdown",
        buttons=buttons,
        x=0.01,  # Left align
        y=1.18,
        xanchor="left",
        yanchor="top",
        showactive=True
    )],
    title={"text": f"{metrics_list[0]} by App", "x": 0.5},  # Center title
    xaxis_title="App",
    yaxis_title=metrics_list[0],
    showlegend=True,
    legend=dict(
        title="Metric",
        orientation="h",
        yanchor="bottom",
        y=1.10,
        xanchor="right",
        x=1
    ),
    margin=dict(t=110)
)

fig.show()
